- [ ] **Step 2: Extend sitemap to emit `<image:image>` per matched URL entry**

Modify `src/app/sitemap.ts`. Required structural change: switch from Next's typed `MetadataRoute.Sitemap` to **raw XML output** because Next's typed Sitemap does not support the `image:image` namespace as of Next 16.

Replace the current default-export pattern with a Route Handler at `src/app/sitemap.ts` returning XML directly:

```ts
import { NextResponse } from 'next/server';
import { SITE_URL } from '@/shared/seo/constants';
import { routing } from '@/i18n/routing';

interface UrlEntry {
  loc: string;          // absolute URL
  lastmod?: string;     // ISO date
  changefreq?: 'always' | 'hourly' | 'daily' | 'weekly' | 'monthly' | 'yearly' | 'never';
  priority?: number;
  images?: { loc: string; title?: string; caption?: string }[];
}

function buildEntries(): UrlEntry[] {
  // ... existing path generation moved here ...
  // For each <url>, optionally attach images: per-page glyph or hero
  // Marketing root: hero PNG/SVG
  // Sign-essay pages: zodiac glyph
}

function escape(s: string): string {
  return s.replace(/&/g, '&amp;').replace(/</g, '&lt;').replace(/>/g, '&gt;').replace(/"/g, '&quot;').replace(/'/g, '&apos;');
}

function xml(entries: UrlEntry[]): string {
  const urlset = entries.map((e) => {
    const images = (e.images ?? []).map((img) => `
    <image:image>
      <image:loc>${escape(img.loc)}</image:loc>${img.title ? `
      <image:title>${escape(img.title)}</image:title>` : ''}${img.caption ? `
      <image:caption>${escape(img.caption)}</image:caption>` : ''}
    </image:image>`).join('');
    return `  <url>
    <loc>${escape(e.loc)}</loc>${e.lastmod ? `
    <lastmod>${e.lastmod}</lastmod>` : ''}${images}
  </url>`;
  }).join('\n');
  return `<?xml version="1.0" encoding="UTF-8"?>
<urlset xmlns="http://www.sitemaps.org/schemas/sitemap/0.9"
        xmlns:image="http://www.google.com/schemas/sitemap-image/1.1">
${urlset}
</urlset>`;
}

export async function GET() {
  const entries = buildEntries();
  return new NextResponse(xml(entries), {
    headers: { 'Content-Type': 'application/xml; charset=utf-8' },
  });
}
```

- [ ] **Step 3: Implement `sun-in-sign-range.ts`**

```ts
import { tropicalToSidereal, getLahiriAyanamsa } from './sidereal';
import { julianDayFromDate } from './julian-day'; // confirm name from T3 review
import { getSunLongitude } from './ephemeris';   // confirm name from T3 review

export type SiderealSign = 'aries' | 'taurus' | 'gemini' | 'cancer' | 'leo' | 'virgo'
                         | 'libra' | 'scorpio' | 'sagittarius' | 'capricorn' | 'aquarius' | 'pisces';

const SIGN_ORDER: SiderealSign[] = [
  'aries', 'taurus', 'gemini', 'cancer', 'leo', 'virgo',
  'libra', 'scorpio', 'sagittarius', 'capricorn', 'aquarius', 'pisces',
];

export type Ayanamsa = 'lahiri';

export interface SunInSignRange {
  sign: SiderealSign;
  start: Date;
  end: Date;
  year: number;
  ayanamsa: Ayanamsa;
}

function siderealSunLongitudeAt(date: Date): number {
  const jd = julianDayFromDate(date);
  const tropical = getSunLongitude(jd);            // adjust per T3 review
  const ay = getLahiriAyanamsa(jd);
  return tropicalToSidereal(tropical, ay);
}

function signFromLongitude(lon: number): SiderealSign {
  return SIGN_ORDER[Math.floor(lon / 30)];
}

function findCrossing(targetSign: SiderealSign, fromUtc: Date, toUtc: Date): Date {
  // Binary search for the moment Sun's sign equals targetSign and prev moment !=
  let lo = fromUtc.getTime();
  let hi = toUtc.getTime();
  while (hi - lo > 60_000) {  // 1-minute precision
    const mid = (lo + hi) / 2;
    const lonAtMid = siderealSunLongitudeAt(new Date(mid));
    const signAtMid = signFromLongitude(lonAtMid);
    if (signAtMid === targetSign) {
      hi = mid;
    } else {
      lo = mid;
    }
  }
  return new Date(hi);
}

export function getSunInSignRange(
  sign: SiderealSign,
  year: number,
  ayanamsa: Ayanamsa = 'lahiri',
): SunInSignRange {
  const yearStart = new Date(Date.UTC(year, 0, 1));
  const yearEnd = new Date(Date.UTC(year + 1, 0, 1));

  // Find all transitions within [year-30d, yearEnd+30d]
  const transitions: Array<{ at: Date; into: SiderealSign }> = [];
  let prevDate = new Date(yearStart.getTime() - 30 * 86_400_000);
  let prevSign = signFromLongitude(siderealSunLongitudeAt(prevDate));
  for (let d = prevDate.getTime() + 86_400_000; d < yearEnd.getTime() + 30 * 86_400_000; d += 86_400_000) {
    const curDate = new Date(d);
    const curSign = signFromLongitude(siderealSunLongitudeAt(curDate));
    if (curSign !== prevSign) {
      const exact = findCrossing(curSign, prevDate, curDate);
      transitions.push({ at: exact, into: curSign });
      prevSign = curSign;
    }
    prevDate = curDate;
  }

  const entry = transitions.find(
    (t) => t.into === sign && t.at.getUTCFullYear() === year,
  );
  if (!entry) {
    throw new Error(`No sun-in-${sign} transition found in year ${year}`);
  }
  const idx = transitions.indexOf(entry);
  const exit = transitions[idx + 1];
  if (!exit) {
    throw new Error(`No exit transition found after sun-in-${sign} ${year}`);
  }
  return { sign, start: entry.at, end: exit.at, year, ayanamsa };
}

export function getSunSignForDate(
  date: Date,
  ayanamsa: Ayanamsa = 'lahiri',
): { sign: SiderealSign; range: SunInSignRange } {
  const lon = siderealSunLongitudeAt(date);
  const sign = signFromLongitude(lon);
  const range = getSunInSignRange(sign, date.getUTCFullYear(), ayanamsa);
  // Handle Dec→Jan windows (e.g., Capricorn): range may have start in prior year
  if (range.end.getTime() < date.getTime() || range.start.getTime() > date.getTime()) {
    const altRange = getSunInSignRange(sign, date.getUTCFullYear() - 1, ayanamsa);
    if (altRange.start.getTime() <= date.getTime() && altRange.end.getTime() > date.getTime()) {
      return { sign, range: altRange };
    }
  }
  return { sign, range };
}
```

**Important:** `getSunLongitude(jd)` is illustrative — astro-eng must use whatever existing function in `chart.ts` / `ephemeris.ts` returns Sun's tropical longitude in degrees from a Julian Day. Adjust import + call to match (per T3 review).

import { NextRequest, NextResponse } from 'next/server';
import { getSunSignForDate } from '@/modules/astro-engine';
import { makeRateLimit } from '@/shared/lib/rate-limit'; // adjust per existing pattern

const rateLimit = makeRateLimit('sidereal-sun-sign', 10, '1 m');

export async function GET(req: NextRequest) {
  const ip = req.headers.get('x-forwarded-for')?.split(',')[0]?.trim() ?? req.headers.get('x-real-ip') ?? 'anonymous';
  const { success } = await rateLimit.limit(ip);
  if (!success) {
    return NextResponse.json({ error: 'rate_limited' }, { status: 429 });
  }

  const { searchParams } = new URL(req.url);
  const dateParam = searchParams.get('date');
  const ayanamsaParam = searchParams.get('ayanamsa') ?? 'lahiri';

  if (!dateParam || !/^\d{4}-\d{2}-\d{2}$/.test(dateParam)) {
    return NextResponse.json({ error: 'invalid_date', detail: 'expected YYYY-MM-DD' }, { status: 400 });
  }
  if (ayanamsaParam !== 'lahiri') {
    return NextResponse.json({ error: 'invalid_ayanamsa', detail: 'only "lahiri" supported' }, { status: 400 });
  }

  const date = new Date(`${dateParam}T12:00:00Z`); // noon UTC to avoid edge timezone effects
  if (Number.isNaN(date.getTime())) {
    return NextResponse.json({ error: 'invalid_date' }, { status: 400 });
  }

  try {
    const { sign, range } = getSunSignForDate(date);
    return NextResponse.json({
      sign,
      startDate: range.start.toISOString(),
      endDate: range.end.toISOString(),
      ayanamsa: range.ayanamsa,
      year: range.year,
    });
  } catch (e) {
    console.error('[sidereal/sun-sign] error', e);
    return NextResponse.json({ error: 'computation_failed' }, { status: 500 });
  }
}
